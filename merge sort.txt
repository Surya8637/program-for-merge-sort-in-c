void mergesort(int low,int high) {
int mid;
if(low<high) {
mid=((low+high)/2);
mergesort(low,mid);
mergesort(mid+1,high);
merge(low,mid,high); }}
void merge(int low,int mid,int high) {
int i,j,h,b[10],k;
